package edu.odu.cs.cs350.pne;

public class classFeatures {


    /**
     * This class Feature searches for words of speech that
     * are at the beginning of a name, after a name
     * or in between the first and last name and returns 1
     * else it returns 0;
     * @param sPeech
     * @return
     *
     */


    public int speechFeatures(String sPeech) {


        if (sPeech.equals("a") || sPeech.equals("an") || sPeech.equals("the")) {

            return 1;
        }

        if (sPeech.equals("and")){

            return 1;
        }

        if(sPeech.equals(",") || sPeech.equals(".") || sPeech.equals("-")){

             return 1;
        }
        else

        return 0;
    }

}
