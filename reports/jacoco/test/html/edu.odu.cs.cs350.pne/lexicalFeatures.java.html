package edu.odu.cs.cs350.pne;

public class lexicalFeatures {

    /** This program checks if the word is an 
     * honorific
     * @param aWord
     * @return
     */
    public char LocateEnglishHonorifics(String aWord)
    {
        switch(aWord){
            case "Dr.": return '1';
            case "Mr.": return '1';
            case "Ms.": return '1';
            case "Mrs.": return '1';
            case "Mx.": return '1';
            case "Madam": return '1';
            case "Sir": return '1';
            case "Esq.": return '1';
            case "Lord": return '1';
            case "Professor": return '1';
            default: return '0';
        }
    }
    
}
